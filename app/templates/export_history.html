from fastapi import APIRouter, Request from fastapi.responses import RedirectResponse from fastapi.templating import Jinja2Templates from app.auth import require_login router = APIRouter() templates = Jinja2Templates(directory="app/templates") @router.get("/export") def export_page(request: Request): if not require_login(request): return RedirectResponse("/login", status_code=303) # IMPORTANT: match your real template filename return templates.TemplateResponse("export_history.html", {"request": request})
